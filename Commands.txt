##### Table Twitter ######

//create table
Create Table Twitter (
date date,
date_time timestamp,
tweet_text string,
type string,
media_type string,
hashtags string,
tweet_url string, 
retweets int)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY '\t'
LINES TERMINATED BY '\n'
STORED AS TEXTFILE;

//Drop Table Twitter

//load data
Load Data Local Inpath 
'/home/cloudera/Downloads/HIVE_Demo/Twitter.txt'
Into Table Twitter;

//see tweet_text where type is 'link'
Select tweet_text, type From Twitter
Where type='text' limit 3;

//see 10 tweet_text order by retweet descending
Select retweets, tweet_text From Twitter 
Order by retweets desc limit 10;

//see hashtags order by date_time
Select date_time, hashtags From Twitter
Where hashtags != ''
Order by date_time;


//see how many tweets per date and per type where media type is not empty
Select date, type, count(*) From Twitter
Where media_type != ''
Group By date, type
Order By date desc;

//see max retweets per date where media type is not empty and max retweets must be more than 10000.
Select date, max(retweets) as mt From Twitter
Where media_type = 'photo'
Group By date
having mt>10000
Order By mt;
